# 获取仓库文件树
curl -s "https://api.github.com/repos/monkeyopen/WestRoad/git/trees/master?recursive=1" | jq '.tree[] | .path'
Curl -s "https://api.github.com/repos/monkeyopen/WestRoad/git/trees/master?recursive=1" | jq -r '.tree[] | "https://raw.githubusercontent.com/monkeyopen/WestRoad/refs/heads/master/\(.path)"'
